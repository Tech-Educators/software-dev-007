CREATE TABLE posts (
    id SERIAL PRIMARY KEY,
    title text,
    content text
);

CREATE TABLE comments (
    id SERIAL PRIMARY KEY,
    username text,
    content text,
    post_id INT REFERENCES posts(id)
)
